34  	public DawidSkene(Set<Category> categories) {
35  
36  		this.objects = new HashMap<String, Datum>();
37  		this.workers = new HashMap<String, Worker>();
38  
39  		this.fixedPriors = false;
40  		this.categories = new HashMap<String, Category>();
41  		for (Category c : categories) {
42  			this.categories.put(c.getName(), c);
43  			if (c.hasPrior()) {
44  				this.fixedPriors = true;
45  			}
46  		}
47  		// We initialize the priors to be uniform across classes
48  		// if the user did not pass any information about the prior values
49  		if (!fixedPriors)
50  			initializePriors();
51  
52  		// By default, we initialize the misclassification costs
53  		// assuming a 0/1 loss function. The costs can be customized
54  		// using the corresponding file
55  		initializeCosts();
56  
57  	}

59  	public void addAssignedLabel(AssignedLabel al) {
60  
61  		String workerName = al.getWorkerName();
62  		String objectName = al.getObjectName();
63  
64  		String categoryName = al.getCategoryName();
65  		assert (this.categories.keySet().contains(categoryName));
66  
67  		// If we already have the object, then just add the label
68  		// in the set of labels for the object.
69  		// If it is the first time we see the object, then create
70  		// the appropriate entry in the objects hashmap
71  		Datum d;
72  		if (this.objects.containsKey(objectName)) {
73  			d = this.objects.get(objectName);
74  		} else {
75  			Set<Category> categories = new HashSet<Category>(this.categories.values());
76  			d = new Datum(objectName, categories);
77  		}
78  		d.addAssignedLabel(al);
79  		this.objects.put(objectName, d);
80  
81  		// If we already have the worker, then just add the label
82  		// in the set of labels assigned by the worker.
83  		// If it is the first time we see the object, then create
84  		// the appropriate entry in the objects hashmap
85  		Worker w;
86  		if (this.workers.containsKey(workerName)) {
87  			w = this.workers.get(workerName);
88  		} else {
89  			Set<Category> categories = new HashSet<Category>(this.categories.values());
90  			w = new Worker(workerName, categories);
91  		}
92  		w.addAssignedLabel(al);
93  		this.workers.put(workerName, w);
94  
95  	}

97  	public void addCorrectLabel(CorrectLabel cl) {
98  
99  		String objectName = cl.getObjectName();
100 		String correctCategory = cl.getCorrectCategory();
101 
102 		Datum d;
103 		if (this.objects.containsKey(objectName)) {
104 			d = this.objects.get(objectName);
105 			d.setGold(true);
106 			d.setCorrectCategory(correctCategory);
107 		} else {
108 			Set<Category> categories = new HashSet<Category>(this.categories.values());
109 			d = new Datum(objectName, categories);
110 			d.setGold(true);
111 			d.setCorrectCategory(correctCategory);
112 		}
113 		this.objects.put(objectName, d);
114 	}

148 	 * TODO: Estimate the model log-likelihood and stop once the log-likelihood values converge

150 	 * @param iterations

152 	public void estimate(int iterations) {
153 
154 		for (int i = 0; i < iterations; i++) {
155 			updateObjectClassProbabilities();
156 			updatePriors();

178 	private HashMap<String, Double> getObjectClassProbabilities(String objectName, String workerToIgnore) {
179 
180 		HashMap<String, Double> result = new HashMap<String, Double>();
181 
182 		Datum d = this.objects.get(objectName);
183 
184 		// If this is a gold example, just put the probability estimate to be
185 		// 1.0
186 		// for the correct class
187 		if (d.isGold()) {
188 			for (String category : this.categories.keySet()) {
189 				String correctCategory = d.getCorrectCategory();
190 				if (category.equals(correctCategory)) {
191 					result.put(category, 1.0);
192 				} else {
193 					result.put(category, 0.0);
194 				}
195 			}
196 			return result;
197 		}
198 
199 		// Let's check first if we have any workers who have labeled this item,
200 		// except for the worker that we ignore
201 		Set<AssignedLabel> labels = d.getAssignedLabels();
202 		if (labels.isEmpty())
203 			return null;
204 		if (workerToIgnore != null && labels.size() == 1) {
205 			for (AssignedLabel al : labels) {
206 				if (al.getWorkerName().equals(workerToIgnore))
207 					return null;
208 			}
209 		}
210 
211 		// If it is not gold, then we proceed to estimate the class
212 		// probabilities using the method of Dawid and Skene and we proceed as
213 		// usual with the M-phase of the EM-algorithm of Dawid&Skene
214 
215 		// Estimate denominator for Eq 2.5 of Dawid&Skene, which is the same
216 		// across all categories
217 		Double denominator = 0.0;
218 
219 		// To compute the denominator, we also compute the nominators across
220 		// all categories, so it saves us time to save the nominators as we
221 		// compute them
222 		HashMap<String, Double> categoryNominators = new HashMap<String, Double>();
223 
224 		for (Category category : categories.values()) {
225 
226 			// We estimate now Equation 2.5 of Dawid & Skene
227 			Double categoryNominator = category.getPrior();
228 
229 			// We go through all the labels assigned to the d object
230 			for (AssignedLabel al : d.getAssignedLabels()) {
231 				Worker w = workers.get(al.getWorkerName());
232 
233 				// If we are trying to estimate the category probability
234 				// distribution
235 				// to estimate the quality of a given worker, then we need to
236 				// ignore
237 				// the labels submitted by this worker.
238 				if (workerToIgnore != null && w.getName().equals(workerToIgnore))
239 					continue;
240 
241 				String assigned_category = al.getCategoryName();
242 				double evidence_for_category = w.getErrorRate(category.getName(), assigned_category);
243 				if (Double.isNaN(evidence_for_category))
244 					continue;
245 				categoryNominator *= evidence_for_category;
246 			}
247 
248 			categoryNominators.put(category.getName(), categoryNominator);
249 			denominator += categoryNominator;
250 		}
251 
252 		for (String category : categories.keySet()) {
253 			Double nominator = categoryNominators.get(category);
254 			if (denominator == 0.0) {
255 				// result.put(category, 0.0);
256 				return null;
257 			} else {
258 				Double probability = Utils.round(nominator / denominator, 5);
259 				result.put(category, probability);
260 			}
261 		}
262 		return result;
263 
264 	}

279 	/**
280 	 * We initialize the misclassification costs using the 0/1 loss
281 	 * 
282 	 * @param engine.getCategories()
283 	 */
284 	private void initializeCosts() {
285 
286 		for (String from : categories.keySet()) {
287 			for (String to : categories.keySet()) {
288 				Category c = categories.get(from);
289 				if (from.equals(to)) {
290 					c.setCost(to, 0.0);
291 				} else {
292 					c.setCost(to, 1.0);
293 				}
294 				categories.put(from, c);
295 			}
296 		}
297 	}
298 
299 	private void initializePriors() {
300 
301 		for (String cat : categories.keySet()) {
302 			Category c = categories.get(cat);
303 			c.setPrior(1.0 / categories.keySet().size());
304 			categories.put(cat, c);
305 		}
306 	}

315 	private void computeEvalConfusionMatrix(Worker w) {
316 		ConfusionMatrix eval_cm = new ConfusionMatrix(this.categories.values());
317 		eval_cm.empty();
318 		for (AssignedLabel l : w.getAssignedLabels()){
319 			
320 			String objectName = l.getObjectName();
321 			Datum d = this.objects.get(objectName);
322 			assert(d != null);
323 			if (!d.isEvaluation()) continue;
324 			
325 			String assignedCategory = l.getCategoryName();
326 			String correctCategory = d.getEvaluationCategory();
327 			
328 			//Double currentCount = eval_cm.getErrorRate(correctCategory, assignedCategory);
329 			eval_cm.addError(correctCategory, assignedCategory, 1.0);
330 		}
331 		eval_cm.normalize();
332 		w.setEvalConfusionMatrix(eval_cm);
333 	}

539 	private void updateObjectClassProbabilities(String objectName) {
540 
541 		Datum d = this.objects.get(objectName);
542 		HashMap<String, Double> probabilities = getObjectClassProbabilities(objectName, null);
543 		if (probabilities == null)
544 			return;
545 		for (String category : probabilities.keySet()) {

547 			d.setCategoryProbability(category, probability);
548 		}
549 		this.objects.put(objectName, d);

552 	/**
553 	 * 
554 	 */
555 	private void updatePriors() {
556 
557 		if (fixedPriors)
558 			return;
559 
560 		HashMap<String, Double> priors = new HashMap<String, Double>();
561 		for (String c : this.categories.keySet()) {
562 			priors.put(c, 0.0);
563 		}
564 
565 		int totalObjects = this.objects.size();
566 		for (Datum d : this.objects.values()) {
567 			for (String c : this.categories.keySet()) {
568 				Double prior = priors.get(c);
569 				Double objectProb = d.getCategoryProbability(c);
570 				prior += objectProb / totalObjects;
571 				priors.put(c, prior);
572 			}
573 		}
574 		setPriors(priors);
575 	}

584 	/**
585 	 * @param lid
586 	 */
587 	private void updateWorkerConfusionMatrix(String workerName) {
588 
589 		Worker w = this.workers.get(workerName);
590 
591 		ConfusionMatrix cm = new ConfusionMatrix(this.categories.values());
592 		cm.empty();
593 
594 		// Scan all objects and change the confusion matrix for each worker
595 		// using the class probability for each object
596 		for (AssignedLabel al : w.getAssignedLabels()) {
597 
598 			// Get the name of the object and the category it
599 			// is classified from this worker.
600 			String objectName = al.getObjectName();
601 			String destination = al.getCategoryName();
602 
603 			// We get the classification of the object
604 			// based on the votes of all the other workers
605 			// We treat this classification as the "correct" one
606 			HashMap<String, Double> probabilities = this.getObjectClassProbabilities(objectName, workerName);
607 			if (probabilities == null)
608 				continue; // No other worker labeled the object
609 
610 			for (String source : probabilities.keySet()) {
611 				Double error = probabilities.get(source);
612 				cm.addError(source, destination, error);
613 			}
614 
615 		}
616 		cm.normalize();
617 
618 		w.setConfusionMatrix(cm);
619
