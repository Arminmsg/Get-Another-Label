22  	public static double getNormalizedP_simple(double q_exp) {
23  
24  		return 0.5 + Math.sqrt(q_exp) / 2;
25  	}

27  	public static double getNormalizedQexp_simple(double p) {
28  
29  		return Math.pow(2 * (p - 0.5), 2);
30  	}

32  	/**
33  	 * Takes as input the probability q of a single worker being correct, and the number of workers k that we use
34  	 * and returns the probability that the majority of the workers will return back the correct answer
35  	 * 
36  	 * @param p
37  	 *          The quality of an individual worker
38  	 * @param k
39  	 *          The number of independent workers
40  	 * @return The probability that the majority of the workers will return back the correct answer
41  	 */
42  	public static Double probabilityCorrect(Double p, Integer k) {
43  
44  		if (k < 0)
45  			return null;
46  		if (k == 0)
47  			return 0.5;
48  
49  		if (p > 1)
50  			return null;
51  		if (p < 0)
52  			return null;
53  
54  		if (p == 0.5)
55  			return 0.5;
56  		if (p < 0.5) {
57  			p = 1 - p;
58  		}
59  
60  		Double result = 0.0;
61  		int low = (int) Math.ceil(0.5 * k + 0.5);
62  		for (int i = low; i <= k; i++) {
63  			// Computing product with logs and then taking the exp, otherwise the Stat.binom over/under-flows.
64  			result += Math.exp(Stat.logBinom(k, i) + i * Math.log(p) + (k - i) * Math.log(1 - p));
65  		}
66  
67  		// In pure math form, we could also multiple this factor with
68  		// ((1/2)*ceil((1/2)*k+1/2)-(1/2)*ceil((1/2)*k))
69  		if (k % 2 == 0) {
70  			result += Math.exp(Stat.logBinom(k, k / 2) + (k / 2) * Math.log(p) + (k / 2) * Math.log(1 - p));
71  		}
72  
73  		return result;
74  	}

76  	public static Double pricingFactor(double from, double to) {
77  
78  		// We do not handle any corner cases (e.g., to==1, or from<=0.5
79  		if (from <= 0.5 || from >= 1)
80  			return null;
81  		if (to <= 0.5 || to >= 1)
82  			return null;
83  		if (from == to)
84  			return 1.0;
85  
86  		// We ensure that from is lower than to. If not, we reverse and return the inverse at the end
87  		boolean reverse = (from > to);
88  		if (reverse) {
89  			double temp = to;
90  			to = from;
91  			from = temp;
92  		}
93  
94  		// First, let's find the minimum number of workers of quality "from" required to reach quality "to"
95  		// We do it very naively here, with a simple, linear search.
96  		// Can be done much more efficiently with a search that increases
97  		// the step over time, but for most cases our loops should return very small
98  		// numbers (less than 50) so it should not be a significant improvement
99  		int minK = 1;
100 		double prob = probabilityCorrect(from, minK);
101 		while (prob < to) {
102 			minK += 2; // No need to increase by 1, as groups of size 2k are the same as 2k-1
103 			prob = probabilityCorrect(from, minK);
104 		}
105 
106 		// The logodds of the probabilityCorrect increase almost linearly with "min_k"
107 		// so we can find a good approximation of the factor by interpolating between
108 		// the values for min_k and min_k-1
109 		double l = Stat.logoggs(probabilityCorrect(from, minK - 2));
110 		double h = Stat.logoggs(probabilityCorrect(from, minK));
111 		double t = Stat.logoggs(to);
112 
113 		// This strange formula is a trivially simple interpolation using the fact that
114 		// the sin of the triangle (l,min_k-2), (h,min_k), (l,min_k)
115 		// is the same as the one in the triangle (l,min_k-2), (t,min_k-2+x), (l,min_k-2+x)
116 		double x = 2 * (t - l) / (h - l);
117 
118 		double result = minK - 2 + x;
119 
120 		if (reverse) {
121 			result = 1 / result;
122 		}
123 		return result;
124 
125 	}

127 	public static void main(String[] args) {
128 
129 		// Let's say that we have a worker with Q_exp = q1 and we pay this worker $1
130 		// How much should we pay someone with Q_exp = q2?
131 		double q1 = 0.36;
132 		double q2 = 0.49;
133

136 		System.out.println("p1=" + p1);
137 		System.out.println("p2=" + p2);
138 		System.out.println("We need to pay $" + pricingFactor(p1, p2));
