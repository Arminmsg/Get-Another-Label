57  	public DawidSkene(Set<Category> categories) {
58  
59  		this.objects = new TreeMap<String, Datum>();
60  		this.workers = new TreeMap<String, Worker>();
61  
62  		this.fixedPriors = false;
63  		this.categories = new HashMap<String, Category>();
64  
65  		for (Category c : categories) {
66  			this.categories.put(c.getName(), c);
67  			if (c.hasPrior()) {
68  				this.fixedPriors = true;
69  			}
70  		}
71  
72  		datumFieldAccessors = FieldAccessors.DATUM_ACCESSORS.getFieldAccessors(this);
73  
74  		// We initialize the priors to be uniform across classes
75  		// if the user did not pass any information about the prior values
76  		if (!fixedPriors)
77  			initializePriors();
78  
79  		// By default, we initialize the misclassification costs
80  		// assuming a 0/1 loss function. The costs can be customized
81  		// using the corresponding file
82  		initializeCosts();
83  	}

85  	public Double getLogLikelihood() {
86  		double result = 0;
87  		
88  		for (Datum d : this.objects.values()) {
89  			for (AssignedLabel al: d.getAssignedLabels()) {
90  				String workerName = al.getWorkerName();
91  				String assignedLabel = al.getCategoryName();
92  				
93  				Map<String, Double> estimatedCorrectLabel = d.getProbabilityVector(ClassificationMethod.DS_Soft);
94  				
95  				for (String from: estimatedCorrectLabel.keySet()) {
96  					Worker w = this.getWorkers().get(workerName);
97  					Double categoryProbability = w.getConfusionMatrix().getErrorRate(from, assignedLabel);
98  					Double labelingProbability = w.getConfusionMatrix().getErrorRate(from, assignedLabel);
99  					result += Math.log(categoryProbability)+Math.log(labelingProbability);
100 				}
101 			}
102 		}
103 		
104 		
105 		return result;
106 	}

108 	public void addAssignedLabel(AssignedLabel al) {
109 
110 		String workerName = al.getWorkerName();
111 		String objectName = al.getObjectName();
112 
113 		String categoryName = al.getCategoryName();
114 		assert (this.categories.keySet().contains(categoryName));
115 
116 		// If we already have the object, then just add the label
117 		// in the set of labels for the object.
118 		// If it is the first time we see the object, then create
119 		// the appropriate entry in the objects hashmap
120 		Datum d;
121 		if (this.objects.containsKey(objectName)) {
122 			d = this.objects.get(objectName);
123 		} else {
124 			d = new Datum(objectName, this);
125 			this.objects.put(objectName, d);
126 		}
127 
128 		d.addAssignedLabel(al);
129 
130 		// If we already have the worker, then just add the label
131 		// in the set of labels assigned by the worker.
132 		// If it is the first time we see the object, then create
133 		// the appropriate entry in the objects hashmap
134 		Worker w;
135 		if (this.workers.containsKey(workerName)) {
136 			w = this.workers.get(workerName);
137 		} else {
138 			w = new Worker(workerName, this);
139 		}
140 		w.addAssignedLabel(al);
141 		this.workers.put(workerName, w);
142 
143 	}

194 	 * in the log likelihood between two consecutive iterations is lower
195 	 * than the specified threshold epsilon, or until executing more than maxIterations
196 	 * 
197 	 * @param maxIterations 
198 	 */
199 	public double estimate(int maxIterations, double epsilon) {
200 		
201 		double pastLogLikelihood = Double.POSITIVE_INFINITY;
202 		double logLikelihood = 0d;
203 		
204 		int cnt = 0;
205 		
206 		
207 		while (cnt <maxIterations && Math.abs(logLikelihood - pastLogLikelihood) > epsilon) {
208 			cnt++;
209 			pastLogLikelihood = getLogLikelihood();
210 			updateObjectClassProbabilities();
211 			updatePriors();
212 			updateWorkerConfusionMatrices();
213 			logLikelihood = getLogLikelihood();
214 		}
215 
216 		datumFieldAccessors = FieldAccessors.DATUM_ACCESSORS.getFieldAccessors(this);
217 		
218 		return logLikelihood;

221 	private HashMap<String, Double> getObjectClassProbabilities(
222 			String objectName, String workerToIgnore) {
223 
224 		HashMap<String, Double> result = new HashMap<String, Double>();
225 
226 		Datum d = this.objects.get(objectName);
227 
228 		// If this is a gold example, just put the probability estimate to be
229 		// 1.0
230 		// for the correct class
231 		if (d.isGold()) {
232 			for (String category : this.categories.keySet()) {
233 				String correctCategory = d.getGoldCategory();
234 				if (category.equals(correctCategory)) {
235 					result.put(category, 1.0);
236 				} else {
237 					result.put(category, 0.0);
238 				}
239 			}
240 			return result;
241 		}
242 
243 		// Let's check first if we have any workers who have labeled this item,
244 		// except for the worker that we ignore
245 		Set<AssignedLabel> labels = d.getAssignedLabels();
246 		if (labels.isEmpty())
247 			return null;
248 		if (workerToIgnore != null && labels.size() == 1) {
249 			for (AssignedLabel al : labels) {
250 				if (al.getWorkerName().equals(workerToIgnore))
251 					return null;
252 			}
253 		}
254 
255 		// If it is not gold, then we proceed to estimate the class
256 		// probabilities using the method of Dawid and Skene and we proceed as
257 		// usual with the M-phase of the EM-algorithm of Dawid&Skene
258 
259 		// Estimate denominator for Eq 2.5 of Dawid&Skene, which is the same
260 		// across all categories
261 		Double denominator = 0.0;
262 
263 		// To compute the denominator, we also compute the nominators across
264 		// all categories, so it saves us time to save the nominators as we
265 		// compute them
266 		HashMap<String, Double> categoryNominators = new HashMap<String, Double>();
267 
268 		for (Category category : categories.values()) {
269 
270 			// We estimate now Equation 2.5 of Dawid & Skene
271 			Double categoryNominator = category.getPrior();
272 
273 			// We go through all the labels assigned to the d object
274 			for (AssignedLabel al : d.getAssignedLabels()) {
275 				Worker w = workers.get(al.getWorkerName());
276 
277 				// If we are trying to estimate the category probability
278 				// distribution
279 				// to estimate the quality of a given worker, then we need to
280 				// ignore
281 				// the labels submitted by this worker.
282 				if (workerToIgnore != null
283 						&& w.getName().equals(workerToIgnore))
284 					continue;
285 
286 				String assigned_category = al.getCategoryName();
287 				double evidence_for_category = w.getErrorRate(
288 						category.getName(), assigned_category);
289 				if (Double.isNaN(evidence_for_category))
290 					continue;
291 				categoryNominator *= evidence_for_category;
292 			}
293 
294 			categoryNominators.put(category.getName(), categoryNominator);
295 			denominator += categoryNominator;
296 		}
297 
298 		for (String category : categories.keySet()) {
299 			Double nominator = categoryNominators.get(category);
300 			if (denominator == 0.0) {
301 				// result.put(category, 0.0);
302 				return null;
303 			} else {
304 				Double probability = Helper.round(nominator / denominator, 5);
305 				result.put(category, probability);
306 			}
307 		}
308 		return result;
309 
310 	}

321 	 * We initialize the misclassification costs using the 0/1 loss
322 	 * 
323 	 * @param engine
324 	 *            .getCategories()
325 	 */
326 	private void initializeCosts() {
327 
328 		for (String from : categories.keySet()) {
329 			for (String to : categories.keySet()) {
330 				Category c = categories.get(from);
331 				if (from.equals(to)) {
332 					c.setCost(to, 0.0);
333 				} else {
334 					c.setCost(to, 1.0);
335 				}
336 				categories.put(from, c);
337 			}
338 		}
339 	}

341 	private void initializePriors() {
342 
343 		for (String cat : categories.keySet()) {
344 			Category c = categories.get(cat);
345 			c.setPrior(1.0 / categories.keySet().size());
346 			categories.put(cat, c);
347 		}
348 	}

358 	private void computeEvalConfusionMatrix(Worker w) {
359 		ConfusionMatrix eval_cm = new ConfusionMatrix(this.categories.values());
360 		eval_cm.empty();
361 		for (AssignedLabel l : w.getAssignedLabels()) {
362 
363 			String objectName = l.getObjectName();
364 			Datum d = this.objects.get(objectName);
365 			assert (d != null);
366 			if (!d.isEvaluation())
367 				continue;
368 
369 			String assignedCategory = l.getCategoryName();
370 			String correctCategory = d.getEvaluationCategory();
371 
372 			// Double currentCount = eval_cm.getErrorRate(correctCategory,
373 			// assignedCategory);
374 			eval_cm.addError(correctCategory, assignedCategory, 1.0);
375 		}
376 		eval_cm.normalize();
377 		w.setEvalConfusionMatrix(eval_cm);
378 	}

418 		Datum d = this.objects.get(objectName);
419 		HashMap<String, Double> probabilities = getObjectClassProbabilities(
420 				objectName, null);

430 	 * 
431 	 */
432 	private void updatePriors() {
433 
434 		if (fixedPriors)
435 			return;
436 
437 		HashMap<String, Double> priors = new HashMap<String, Double>();
438 		for (String c : this.categories.keySet()) {
439 			priors.put(c, 0.0);
440 		}
441 
442 		int totalObjects = this.objects.size();
443 		for (Datum d : this.objects.values()) {
444 			for (String c : this.categories.keySet()) {
445 				Double prior = priors.get(c);
446 				Double objectProb = d.getCategoryProbability(
447 						Datum.ClassificationMethod.DS_Soft, c);
448 				prior += objectProb / totalObjects;
449 				priors.put(c, prior);
450 			}
451 		}
452 		setPriors(priors);
453 	}

462 	/**
463 	 * @param lid
464 	 */
465 	private void updateWorkerConfusionMatrix(String workerName) {
466 
467 		Worker w = this.workers.get(workerName);
468 
469 		ConfusionMatrix cm = new ConfusionMatrix(this.categories.values());
470 		cm.empty();
471 
472 		// Scan all objects and change the confusion matrix for each worker
473 		// using the class probability for each object
474 		for (AssignedLabel al : w.getAssignedLabels()) {
475 
476 			// Get the name of the object and the category it
477 			// is classified from this worker.
478 			String objectName = al.getObjectName();
479 			String destination = al.getCategoryName();
480 
481 			// We get the classification of the object
482 			// based on the votes of all the other workers
483 			// We treat this classification as the "correct" one
484 			HashMap<String, Double> probabilities = this
485 					.getObjectClassProbabilities(objectName, workerName);
486 			if (probabilities == null)
487 				continue; // No other worker labeled the object
488 
489 			for (String source : probabilities.keySet()) {
490 				Double error = probabilities.get(source);
491 				cm.addError(source, destination, error);
492 			}
493 
494 		}
495 		cm.normalize();
496 
497 		w.setConfusionMatrix(cm);
498 
499 	}
