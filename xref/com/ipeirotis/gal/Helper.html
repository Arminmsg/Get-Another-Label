30  	/**
31  	 * Gets as input a "soft label" (i.e., a distribution of probabilities over
32  	 * classes) and returns the expected cost of this soft label.
33  	 * 
34  	 * @param p
35  	 * @return The expected cost of this soft label
36  	 */
37  	public static Double getExpectedSoftLabelCost(Map<String, Double> probabilities, Map<String, Category>	categories) {
38  
39  		Double c = 0.0;
40  		for (String c1 : probabilities.keySet()) {
41  			for (String c2 : probabilities.keySet()) {
42  				Double p1 = probabilities.get(c1);
43  				Double p2 = probabilities.get(c2);
44  				Double cost = categories.get(c1).getCost(c2);
45  				c += p1 * p2 * cost;
46  			}
47  		}
48  
49  		return c;
50  	}

52  	/**
53  	 * Gets as input a "soft label" (i.e., a distribution of probabilities over
54  	 * classes) and returns the smallest possible cost for this soft label.
55  	 * 
56  	 * @param p
57  	 * @return The classification cost of this soft label if we classify in the class that has the minimum exp cost.
58  	 */
59  	public static Double getMinCostLabelCost(Map<String, Double> probabilities, Map<String, Category>	categories) {
60  
61  		Double cost  = 0.0;
62  
63  		// We know that the classification cost is minimized if we pick *one* category, as opposed to a mixture.
64  		// What is the category that has the lowest cost in this case?
65  		String to = getMinCostLabel(probabilities, categories);
66  		if (to == null) {
67  			return Double.NaN;
68  		}
69  		
70  			for (String from : probabilities.keySet()) {
71  				// With probability p it actually belongs to class from
72  				Double p = probabilities.get(from);
73  				Double c = categories.get(from).getCost(to);
74  				
75  				cost += p * c;
76  			}
77  
78  		return cost;
79  	}

81  	public static String getMinCostLabel(Map<String, Double> softLabel, Map<String, Category>	categories) {
82  
83  		String result = null;
84  		Double min_cost = Double.MAX_VALUE;
85  
86  		for (String c1 : softLabel.keySet()) {
87  			// So, with probability p1 it belongs to class c1
88  			// Double p1 = probabilities.get(c1);
89  
90  			// What is the cost in this case?
91  			Double costfor_c1 = 0.0;
92  			for (String c2 : softLabel.keySet()) {
93  				// With probability p2 it actually belongs to class c2
94  				Double p2 = softLabel.get(c2);
95  				Double cost = categories.get(c2).getCost(c1);
96  				costfor_c1 += p2 * cost;
97  
98  			}
99  
100 			if (costfor_c1 < min_cost) {
101 				result = c1;
102 				min_cost = costfor_c1;
103 			}
104 
105 		}
106 
107 		return result;
108 	}

110 	public static String getMaxLikelihoodLabel(Map<String, Double> probabilities, Map<String, Category> categories) {
111 
112 		String result = null;
113 		double maxProbability = -1;
114 
115 		for (String category : probabilities.keySet()) {
116 			Double probability = probabilities.get(category);
117 			if (probability > maxProbability) {
118 				maxProbability = probability;
119 				result = category;
120 			} else if (probability == maxProbability) {
121 				// In case of a tie, break ties randomly
122 				// TODO: This is a corner case. We can also break ties
123 				// using the priors. But then we also need to group together
124 				// all the ties, and break ties probabilistically across the
125 				// group. Otherwise, we slightly favor the later comparisons.
126 				if (Math.random() > 0.5) {
127 					maxProbability = probability;
128 					result = category;
129 				}
130 			}
131 		}
132 
133 		return result;
134 	}

138 	 * Gets as input a "soft label" (i.e., a distribution of probabilities over
139 	 * classes) and returns the expected cost for the maximum likelihood label.
140 	 * 
141 	 * @param p
142 	 * @return The classification cost of this soft label if we classify in the max likelihood class
143 	 */
144 	public static Double getMaxLikelihoodCost(Map<String, Double> probabilities, Map<String, Category>	categories) {
145 
146 		Double cost = 0.0;
147 
148 		String to = getMaxLikelihoodLabel(probabilities, categories);
149 		if (to == null) {
150 			return Double.NaN;
151 		}
152 			
153 			for (String from : probabilities.keySet()) {
154 				// With probability p it actually belongs to class from
155 				Double p = probabilities.get(from);
156 				Double c = categories.get(from).getCost(to);
157 				cost += p * c;
158 			}
159 
160 		return cost;
161 	}

167 	 * completely random labels.
168 	 * 
169 	 * @return The expected cost of a spammer worker
170 	 */
171 	public static Double getMinSpammerCost(Map<String, Category>	categories) {
172 
173 		Map<String, Double> prior = new HashMap<String, Double>();
174 		for (Category c : categories.values()) {
175 			prior.put(c.getName(), c.getPrior());

196 	public static String readFile(String FileName) {
197 	
198 		StringBuffer buffer = new StringBuffer();
199 	
200 		try {
201 			BufferedReader dataInput = new BufferedReader(new FileReader(new File(FileName)));
202 			String line;
203 	
204 			while ((line = dataInput.readLine()) != null) {
205 				// buffer.append(cleanLine(line.toLowerCase()));
206 				buffer.append(line);
207 				buffer.append('\n');
208 			}
209 			dataInput.close();
210 		} catch (Exception ex) {
211 			ex.printStackTrace();
212 		}
213 		return buffer.toString();
214 	}

216 	public static Double round(double d, int decimalPlace) {
217 	
218 		// see the Javadoc about why we use a String in the constructor
219 		// http://java.sun.com/j2se/1.5.0/docs/api/java/math/BigDecimal.html#BigDecimal(double)
220 		BigDecimal bd = new BigDecimal(Double.toString(d));
221 		bd = bd.setScale(decimalPlace, BigDecimal.ROUND_HALF_UP);
222 		return bd.doubleValue();
223 	}
